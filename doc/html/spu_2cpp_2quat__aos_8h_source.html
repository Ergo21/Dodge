78 inline const Quat slerp( float t, Quat unitQuat0, Quat unitQuat1 )
79 {
80  Quat start;
81  vec_float4 scales, scale0, scale1, cosAngle, angle, tttt, oneMinusT, angles, sines;
82  vec_uint4 selectMask;
83  vec_uchar16 shuffle_xxxx = (vec_uchar16)spu_splats((int)0x00010203);
84  vec_uchar16 shuffle_yyyy = (vec_uchar16)spu_splats((int)0x04050607);
85  vec_uchar16 shuffle_zzzz = (vec_uchar16)spu_splats((int)0x08090a0b);
86  cosAngle = _vmathVfDot4( unitQuat0.get128(), unitQuat1.get128() );
87  cosAngle = spu_shuffle( cosAngle, cosAngle, shuffle_xxxx );
88  selectMask = (vec_uint4)spu_cmpgt( spu_splats(0.0f), cosAngle );
89  cosAngle = spu_sel( cosAngle, negatef4( cosAngle ), selectMask );
90  start = Quat( spu_sel( unitQuat0.get128(), negatef4( unitQuat0.get128() ), selectMask ) );
91  selectMask = (vec_uint4)spu_cmpgt( spu_splats(_VECTORMATH_SLERP_TOL), cosAngle );
92  angle = acosf4( cosAngle );
93  tttt = spu_splats(t);
94  oneMinusT = spu_sub( spu_splats(1.0f), tttt );
95  angles = spu_sel( spu_splats(1.0f), oneMinusT, (vec_uint4)spu_maskb(0x0f00) );
96  angles = spu_sel( angles, tttt, (vec_uint4)spu_maskb(0x00f0) );
97  angles = spu_mul( angles, angle );
98  sines = sinf4( angles );
99  scales = divf4( sines, spu_shuffle( sines, sines, shuffle_xxxx ) );
100  scale0 = spu_sel( oneMinusT, spu_shuffle( scales, scales, shuffle_yyyy ), selectMask );
101  scale1 = spu_sel( tttt, spu_shuffle( scales, scales, shuffle_zzzz ), selectMask );
102  return Quat( spu_madd( start.get128(), scale0, spu_mul( unitQuat1.get128(), scale1 ) ) );
103 }

105 inline const Quat squad( float t, Quat unitQuat0, Quat unitQuat1, Quat unitQuat2, Quat unitQuat3 )
106 {

275 inline const Quat Quat::rotation( Vector3 unitVec0, Vector3 unitVec1 )
276 {
277  Vector3 crossVec;
278  vec_float4 cosAngle, cosAngleX2Plus2, recipCosHalfAngleX2, cosHalfAngleX2, res;
279  cosAngle = _vmathVfDot3( unitVec0.get128(), unitVec1.get128() );
280  cosAngle = spu_shuffle( cosAngle, cosAngle, (vec_uchar16)spu_splats(0x00010203) );
281  cosAngleX2Plus2 = spu_madd( cosAngle, spu_splats(2.0f), spu_splats(2.0f) );
282  recipCosHalfAngleX2 = rsqrtf4( cosAngleX2Plus2 );
283  cosHalfAngleX2 = spu_mul( recipCosHalfAngleX2, cosAngleX2Plus2 );
284  crossVec = cross( unitVec0, unitVec1 );
285  res = spu_mul( crossVec.get128(), recipCosHalfAngleX2 );
286  res = spu_sel( res, spu_mul( cosHalfAngleX2, spu_splats(0.5f) ), (vec_uint4)spu_maskb(0x000f) );
287  return Quat( res );
288 }

329 inline const Quat Quat::operator *( Quat quat ) const
330 {
331  vec_float4 ldata, rdata, qv, tmp0, tmp1, tmp2, tmp3;
332  vec_float4 product, l_wxyz, r_wxyz, xy, qw;
333  ldata = mVec128;
334  rdata = quat.mVec128;
335  vec_uchar16 shuffle_wwww = (vec_uchar16)spu_splats((int)0x0c0d0e0f);
336  tmp0 = spu_shuffle( ldata, ldata, _VECTORMATH_SHUF_YZXW );
337  tmp1 = spu_shuffle( rdata, rdata, _VECTORMATH_SHUF_ZXYW );
338  tmp2 = spu_shuffle( ldata, ldata, _VECTORMATH_SHUF_ZXYW );
339  tmp3 = spu_shuffle( rdata, rdata, _VECTORMATH_SHUF_YZXW );
340  qv = spu_mul( spu_shuffle( ldata, ldata, shuffle_wwww ), rdata );
341  qv = spu_madd( spu_shuffle( rdata, rdata, shuffle_wwww ), ldata, qv );
342  qv = spu_madd( tmp0, tmp1, qv );
343  qv = spu_nmsub( tmp2, tmp3, qv );
344  product = spu_mul( ldata, rdata );
345  l_wxyz = spu_rlqwbyte( ldata, 12 );
346  r_wxyz = spu_rlqwbyte( rdata, 12 );
347  qw = spu_nmsub( l_wxyz, r_wxyz, product );
348  xy = spu_madd( l_wxyz, r_wxyz, product );
349  qw = spu_sub( qw, spu_rlqwbyte( xy, 8 ) );
350  return Quat( spu_sel( qv, qw, (vec_uint4)spu_maskb( 0x000f ) ) );
351 }

359 inline const Vector3 rotate( Quat quat, Vector3 vec )
360 {
361  vec_float4 qdata, vdata, product, tmp0, tmp1, tmp2, tmp3, wwww, qv, qw, res;
362  qdata = quat.get128();
363  vdata = vec.get128();
364  vec_uchar16 shuffle_xxxx = (vec_uchar16)spu_splats((int)0x00010203);
365  vec_uchar16 shuffle_wwww = (vec_uchar16)spu_splats((int)0x0c0d0e0f);
366  tmp0 = spu_shuffle( qdata, qdata, _VECTORMATH_SHUF_YZXW );
367  tmp1 = spu_shuffle( vdata, vdata, _VECTORMATH_SHUF_ZXYW );
368  tmp2 = spu_shuffle( qdata, qdata, _VECTORMATH_SHUF_ZXYW );
369  tmp3 = spu_shuffle( vdata, vdata, _VECTORMATH_SHUF_YZXW );
370  wwww = spu_shuffle( qdata, qdata, shuffle_wwww );
371  qv = spu_mul( wwww, vdata );
372  qv = spu_madd( tmp0, tmp1, qv );
373  qv = spu_nmsub( tmp2, tmp3, qv );
374  product = spu_mul( qdata, vdata );
375  qw = spu_madd( spu_rlqwbyte( qdata, 4 ), spu_rlqwbyte( vdata, 4 ), product );
376  qw = spu_add( spu_rlqwbyte( product, 8 ), qw );
377  tmp1 = spu_shuffle( qv, qv, _VECTORMATH_SHUF_ZXYW );
378  tmp3 = spu_shuffle( qv, qv, _VECTORMATH_SHUF_YZXW );
379  res = spu_mul( spu_shuffle( qw, qw, shuffle_xxxx ), qdata );
380  res = spu_madd( wwww, qv, res );
381  res = spu_madd( tmp0, tmp1, res );
382  res = spu_nmsub( tmp2, tmp3, res );
383  return Vector3( res );
384 }
