139 void Controller::moveX(int d) {
140  float x = 1; //1 is right
141  float z = 0; //1 is Forward, angle left is positive going to pi before flipping to -pi
142 
143  if(tAngMem >= 0 && tAngMem <= pi/2) {
144  x = 1-tAngMem/(pi/2);
145  z = 1 - x;
146  }
147  else if(tAngMem > pi/2 && tAngMem <= pi) {
148  x = -(tAngMem-(pi/2))/(pi/2); //Neg 0 -1
149  z = 1+x; //Pos 1 0
150  }
151  else if(tAngMem > -pi && tAngMem <= -pi/2) {
152  x = (tAngMem + (pi/2))/(pi/2); //Neg 0 -1
153  z = -1-x; //Neg -1 0
154  }
155  else {
156  x = 1+tAngMem/(pi/2); //Pos 1 0
157  z = x-1; //Neg 0 -1
158  }
159 
160  Vector4 thTemCam = Camera::getInstance().getCameraM().getCol3();
161 
162  x = x*d;
163  z = z*d;
164 
165  if(conPhy->movCol(conData->getPlayer(), 'X', x)) {
166  x = 0;
167  }
168  else {
169  conData->getPlayer()->moveX(x);
170  }
171 
172  if(conPhy->movCol(conData->getPlayer(), 'Z', z)) {
173  z = 0;
174  }
175  else {
176  conData->getPlayer()->moveZ(z);
177  }
178 
179  thTemCam.setX(thTemCam.getX()+x);
